#include <windows.h>  // for MS Windows
#include <GL/glut.h>  // GLUT, include glu.h and gl.h

/* Handler for window-repaint event. Call back when the window first appears and
whenever the window needs to be re-painted. */
void display() {
	glClearColor(1.0f, 1.0f, 1.0f, 1.0f); // Set background color to black and opaque
	glClear(GL_COLOR_BUFFER_BIT);         // Clear the color buffer (background)

	//POINT Drawing

    glPointSize(10);

    glBegin(GL_POINTS);

    	//drawing the blue point in co-or (0.5,0.5)
        glColor3f(0,0,1);
        glVertex2f(0.5 ,0.5);

	   //drawing the yellow point in co-or (0,0)
        glColor3f(1,1,0);
        glVertex2f(0 ,0);

    glEnd();

    //LINE Drawing

    glBegin(GL_LINES);

        glColor3f(0,0,0);

        //X-Axis
        //Drawing a line with vertex (-1,0) and (1,0)
        glVertex2f(-1,0);
        glVertex2f(1,0);

        //Y-Axis
        //drawing a line with vertex (0,-1) and (0,1)
        glVertex2f(0,-1);
        glVertex2f(0,1);

    glEnd();

    //TRIANGLE Drawing

    glBegin(GL_TRIANGLES);

        glColor3f(0,1,0);
        glVertex2f(-0.6, -0.2);
        glVertex2f(-0.6, -0.5);
        glVertex2f(-0.3, -0.5);

    glEnd();

    //RECTANGLE Drawing

    glBegin(GL_QUADS);

        glColor3f(1,0,0);
        glVertex2f(-1.0, 0.9);
        glVertex2f(-1.0, 0.6);
        glVertex2f(-0.5, 0.6);
        glVertex2f(-0.5, 0.9);

    glEnd();

    //RECTANGLE Drawing

    glBegin(GL_QUADS);

        glColor3ub(153,217,234);
        glVertex2f(-0.4, 0.7);
        glVertex2f(-0.4, 0.1);
        glVertex2f(-0.1, 0.1);
        glVertex2f(-0.1, 0.7);

    glEnd();

    //PENTAGON Drawing

    glBegin(GL_POLYGON);

        glColor3f(1,0,1);
        glVertex2f(0.65, -0.1);
        glVertex2f(0.35, -0.3);
        glVertex2f(0.48, -0.65);
        glVertex2f(0.8, -0.65);
        glVertex2f(0.93, -0.3);

    glEnd();

	glFlush();  // Render now
}

/* Main function: GLUT runs as a console application starting at main()  */
int main(int argc, char** argv) {
	glutInit(&argc, argv);                 // Initialize GLUT
	glutCreateWindow("Welcome TO CG Section-F"); // Create a window with the given title
	glutInitWindowSize(320, 320);   // Set the window's initial width & height
	glutDisplayFunc(display); // Register display callback handler for window re-paint
	glutMainLoop();           // Enter the event-processing loop
	return 0;
}
