#include <windows.h>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <GL/gl.h>
#include <GL/glut.h>
using namespace std;
 
float a1,b1,a2,b2,c1,d1,c2,d2;
 
void drawAxes(){
    glColor3f(0,0,0);
    glBegin(GL_LINES);
    glVertex2i(-100,0); glVertex2i(100,0);
    glVertex2i(0,-100); glVertex2i(0,100);
    glEnd();
}
 
void dda(float X1,float Y1,float X2,float Y2,int r,int g,int b){
    float m=(Y2-Y1)/(X2-X1);
    glColor3ub(r,g,b);
    glBegin(GL_POINTS);
    if(X1>X2){swap(X1,X2);swap(Y1,Y2);m=-m;}
    if(Y1>Y2){swap(Y1,Y2);m=-m;}
    if(m<=1){
        while(X1<X2 && Y1<Y2){
            X1=X1+1;
            Y1=Y1+m;
            glVertex2i(round(X1),round(Y1));
        }
    }else{
        while(X1<X2 && Y1<Y2){
            X1=X1+(1/m);
            Y1=Y1+1;
            glVertex2i(round(X1),round(Y1));
        }
    }
    glEnd();
}
 
void display(){
    glClear(GL_COLOR_BUFFER_BIT);
    drawAxes();
    dda(a1,b1,a2,b2,255,0,0);
    dda(c1,d1,c2,d2,0,255,0);
    glFlush();
}
 
void init(){
    glClearColor(1,1,1,0);
    glPointSize(4);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(-100,100,-100,100);
}
 
int main(int c,char** v){
    cout<<"Line 1 (X1 Y1 X2 Y2): "; cin>>a1>>b1>>a2>>b2;
    cout<<"Line 2 (X1 Y1 X2 Y2): "; cin>>c1>>d1>>c2>>d2;
    glutInit(&c,v);
    glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB);
    glutInitWindowSize(600,600);
    glutCreateWindow("DDA");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
